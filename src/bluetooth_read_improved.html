int i = 0;
char blueToothVal[10] = "";           //value sent over via bluetooth

void setup() {
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    //if there is data being recieved
    blueToothVal[i] = Serial.read(); //read it
    if (blueToothVal[i] == 'S') {
      i = 0;
    } else {
      i++;
    }

  }

  if (i == 8) {
    i = 0;
    Serial.write(blueToothVal);
  }
}
