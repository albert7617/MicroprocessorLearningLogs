int A = 2;
int B = 4;
int C = 7;
int switch_ = 0;

void setup() {
  pinMode (A, OUTPUT);
  pinMode (B, OUTPUT);
  pinMode (C, OUTPUT);
}

void loop() {
  switch(switch_) {
    case 0:
      digitalWrite(A, HIGH);
      digitalWrite(B, HIGH);
      digitalWrite(C, LOW);
      switch_=1;
      break;
    case 1:
      digitalWrite(A, LOW);
      digitalWrite(B, LOW);
      digitalWrite(C, HIGH);
      switch_=2;
      break;
    case 2:
      digitalWrite(A, HIGH);
      digitalWrite(B, LOW);
      digitalWrite(C, HIGH);
      switch_=0;
      break;
    default:
      break;
  }
}
