char blueToothVal;

void setup() {
  Serial.begin(9600);
  pinMode(13,OUTPUT);
}

void loop() {
  if(Serial.available()) {
    blueToothVal=Serial.read();
  }
  if (blueToothVal=='n') {
    digitalWrite(13,HIGH);
  } else if (blueToothVal=='f') {
    digitalWrite(13,LOW);
  }
  delay(1000);
}
