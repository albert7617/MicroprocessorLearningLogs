int speedCtrl = 3;  // PWM output
int cw = 9;         // L298N direction control clockwise
int ccw = 10;       // L298N direction control counterclockwise

void setup() {
  // Setting DC mortor spinning direction to clockwise
  digitalWrite(cw, HIGH); 
  digitalWrite(ccw, LOW);
}

void loop() {
  analogWrite(speedCtrl, 128); // Output a 50% duty-cycle PWM signal
}
